"""
AutoReply Module for Telegram Userbot
Fitur: monitoring discussion groups, auto-reply dengan wording custom
"""

import logging
import os
from typing import List, Dict
from telethon import events
from motor.motor_asyncio import AsyncIOMotorClient
import asyncio
from datetime import datetime

logger = logging.getLogger(__name__)


class AutoReply:
    """Module untuk AutoReply Discussion Group"""

    def __init__(self, client, db):
        self.client = client
        self.db = db
        self.autoreply_groups = []
        self.wordings = []

        # Userbot info (initialized lazily)
        self.userbot_id = None
        self.userbot_username = None

    async def _init_userbot_info(self):
        """Initialize userbot info if not already set"""
        if self.userbot_id is None:
            try:
                me = await self.client.get_me()
                self.userbot_id = me.id
                self.userbot_username = me.username
                logger.info(f"Userbot initialized: ID={self.userbot_id}, username={self.userbot_username}")
            except Exception as e:
                logger.error(f"Failed to initialize userbot info: {e}")
                self.userbot_id = None
                self.userbot_username = None

    async def load_config(self):
        """Load config dari MongoDB"""
        logger.info("Loading AutoReply config...")

        # Load autoreply groups
        groups = await self.db.autoreply_groups.find().to_list(None)
        self.autoreply_groups = [str(g['group_id']) for g in groups]
        logger.info(f"Loaded {len(self.autoreply_groups)} autoreply groups")

        # Load wordings
        wordings = await self.db.wordings.find().to_list(None)
        self.wordings = [
            {
                'id': str(w['id']),
                'send_count': w.get('send_count', 1),
                'keyword': w.get('keyword', ''),
                'message': w.get('message', ''),
                'group_id': str(w.get('group_id', 'all'))
            }
            for w in wordings
        ]
        logger.info(f"Loaded {len(self.wordings)} wordings")

    async def save_group(self, group_id: str):
        """Simpan group ke database"""
        try:
            await self.db.autoreply_groups.update_one(
                {'group_id': group_id},
                {'$setOnInsert': {
                    'group_id': group_id,
                    'created_at': datetime.utcnow()
                }},
                upsert=True
            )
            if group_id not in self.autoreply_groups:
                self.autoreply_groups.append(str(group_id))
            return True
        except Exception as e:
            logger.error(f"Gagal save group: {e}")
            return False

    async def delete_group(self, group_id: str):
        """Hapus group dari database"""
        try:
            await self.db.autoreply_groups.delete_one({'group_id': group_id})
            if group_id in self.autoreply_groups:
                self.autoreply_groups.remove(str(group_id))
            return True
        except Exception as e:
            logger.error(f"Gagal delete group: {e}")
            return False

    async def save_wording(self, data: Dict):
        """Simpan wording baru"""
        try:
            import uuid
            data['id'] = str(uuid.uuid4())

            result = await self.db.wordings.insert_one(data)

            self.wordings.append({
                'id': str(result.inserted_id),
                'send_count': data.get('send_count', 1),
                'keyword': data.get('keyword', ''),
                'message': data.get('message', ''),
                'group_id': data.get('group_id', 'all')
            })
            return True, str(result.inserted_id)
        except Exception as e:
            logger.error(f"Gagal save wording: {e}")
            return False, str(e)

    async def delete_wording(self, wording_id: str):
        """Hapus wording berdasarkan ID"""
        try:
            await self.db.wordings.delete_one({'id': wording_id})
            self.wordings = [
                w for w in self.wordings
                if w['id'] != wording_id
            ]
            return True
        except Exception as e:
            logger.error(f"Gagal delete wording: {e}")
            return False

    async def list_wordings(self):
        """List semua wording aktif"""
        if not self.wordings:
            return "Belum ada wording aktif."

        result = []
        for i, w in enumerate(self.wordings):
            group_tag = w['group_id'] if w['group_id'] != 'all' else "All Groups"
            result.append(
                f"<code>{i+1}</code>. <b>Keyword:</b> <code>{w['keyword']}</b>\n"
                f"   <b>Group:</b> {group_tag}\n"
                f"   <b>Send Count:</b> {w['send_count']}\n"
                f"   <b>Message:</b> {w['message'][:60]}..."
            )

        return "\n\n".join(result)

    @staticmethod
    def parse_arguments(args: str) -> Dict:
        """Parse argument dari userbot command"""
        parts = args.strip().split(maxsplit=1)
        return {
            'group_id': parts[0] if len(parts) > 0 else None,
            'keyword': None,
            'message': None,
            'send_count': None
        }

    async def get_user_input_interactive(self, msg, update):
        """Tanya input ke user secara interaktif"""
        from telethon.tl.functions.messages import GetDialogsRequest

        # Tanya jumlah pengiriman
        await msg.respond("📦 <b>Berapa kali bot harus mengirim pesan auto-reply?</b>")
        send_count = await self.get_text_input(update, msg)

        # Tanya kata kunci
        await msg.respond("🔑 <b>Masukkan kata kunci:</b>")
        keyword = await self.get_text_input(update, msg)

        # Tanya pesan auto-reply
        await msg.respond("💬 <b>Pesan auto-reply:</b>")
        message = await self.get_text_input(update, msg)

        return {
            'send_count': send_count,
            'keyword': keyword,
            'message': message
        }

    async def get_text_input(self, update, msg=None):
        """Get text input dari user"""
        update_text = update.message.text if update.message else ""
        if update_text.startswith("/") and msg:
            # Ambil text dari pesan terakhir
            return msg.text
        return update_text

    async def send_link_to_saved_messages(self, client, message_id: int, channel_id: str, msg: str):
        """Kirim link ke Saved Messages"""
        try:
            from telethon.tl.functions.messages import GetHistoryRequest
            from telethon.tl.tl_schema import TL

            link = f"https://t.me/c/{channel_id}/{message_id}"

            # Send link ke Saved Messages
            try:
                # Cari saved messages dialog
                dialogs = await self.client.get_dialogs()
                saved_dialogs = [d for d in dialogs if d.is_user and d.entity.id == self.client.user_id]

                if saved_dialogs:
                    saved_msg = saved_dialogs[0]
                    link_text = f"🔗 <b>Link Pesan Auto-Reply:</b>\n{link}\n\n<b>Original Message:</b>\n{msg[:500]}"
                    await self.client.send_message(saved_msg.entity, link_text)
                else:
                    link_text = f"🔗 <b>Link Pesan Auto-Reply:</b>\n{link}\n\n<b>Original Message:</b>\n{msg[:500]}"
                    await self.client.send_message("me", link_text)

            except Exception as e:
                # Fallback: kirim ke 'me' (saved messages) directly
                link_text = f"🔗 <b>Link Pesan Auto-Reply:</b>\n{link}\n\n<b>Original Message:</b>\n{msg[:500]}"
                await self.client.send_message("me", link_text)

            logger.info(f"Link dikirim ke Saved Messages: {link}")
            return True
        except Exception as e:
            logger.error(f"Gagal kirim ke Saved Messages: {e}")
            return False

    async def monitor_and_reply(self, message):
        """Monitor pesan dan kirim auto-reply"""
        if message.sender_id is None:  # Ignore bot messages
            return

        try:
            message_text = message.message or ""

            # Cari matching wording
            for wording in self.wordings:
                keyword = wording.get('keyword', '').lower()
                if keyword and keyword in message_text.lower():
                    # Cek group match
                    if wording['group_id'] != 'all':
                        if str(message.chat_id) != str(wording['group_id']):
                            continue

                    # Kirim auto-reply
                    channel_id = str(message.chat_id)

                    reply_count = int(wording.get('send_count', 1))
                    for i in range(reply_count):
                        try:
                            await self.client.send_message(message.chat_id, wording['message'])
                            await asyncio.sleep(5)  # Delay antar reply

                            # Kirim link ke Saved Messages setelah reply pertama
                            if i == 0:
                                await self.send_link_to_saved_messages(self.client, message.id, channel_id, message_text)

                            logger.info(f"Auto-reply dikirim ke group {channel_id}")
                        except Exception as e:
                            logger.error(f"Gagal kirim reply: {e}")
                            break
                    break  # Hanya reply dengan satu wording (bisa expand ke multiple)
        except Exception as e:
            logger.error(f"Error dalam monitor_and_reply: {e}")
            pass
